üìö Component System

A React-like component system for building reusable, stateful UI elements. 

// Define a component
class Counter extends Component {
    constructor(props) {
        super(props);
        this.state = { count: 0 };
    }

    increment() {
        this.setState({ count: this.state.count + 1 });
    }

    render() {
        return h('div', {}, [
            h('h2', {}, `Count: ${this.state.count}`),
            h('button', { onClick: () => this.increment() }, 'Increment')
        ]);
    }
}

// Mount to DOM
new Counter().mount('#app');

üî¢ Interactive Counter (Stateful)

class Counter extends Component {
    constructor(props) {
        super(props);
        this.state = { count: 0 };
    }

    render() {
        return h('div', { style: { textAlign: 'center' } }, [
            h('h1', {}, `Count: ${this.state.count}`),
            h('button', { onClick: () => this.setState({ count: this.state.count + 1 }) }, '+'),
            h('button', { onClick: () => this.setState({ count: this.state.count - 1 }) }, '-')
        ]);
    }
}
